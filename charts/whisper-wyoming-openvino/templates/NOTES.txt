Wyoming API is now available and can be accessed:

{{- if eq .Values.wyomingApi.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "whisper-wyoming-openvino.fullname" . }}-wyoming-api)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Wyoming API URL: tcp://$NODE_IP:$NODE_PORT"
{{- else if eq .Values.wyomingApi.service.type "LoadBalancer" }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "whisper-wyoming-openvino.fullname" . }}-wyoming-api'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "whisper-wyoming-openvino.fullname" . }}-wyoming-api --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Wyoming API URL: tcp://$SERVICE_IP:{{ .Values.wyomingApi.service.port }}"
{{- else if eq .Values.wyomingApi.service.type "ClusterIP" }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "{{ include "whisper-wyoming-openvino.wyomingApiSelectorLabels" . | replace "\n" "," }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Forward local port to Wyoming API:"
  echo "  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME {{ .Values.wyomingApi.service.port }}:{{ .Values.wyomingApi.service.port }}"
  echo "Then access at: tcp://127.0.0.1:{{ .Values.wyomingApi.service.port }}"
{{- end }}
